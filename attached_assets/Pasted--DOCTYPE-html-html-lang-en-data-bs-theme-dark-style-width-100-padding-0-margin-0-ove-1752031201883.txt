<script>
// Константы URL для API
const URLS = {
    bind_device: "/bind_device",
    get_remote_devices: "/get_remote_devices",
    attach_device: "/attach_device",
    detach_device: "/detach_device"
}

// Обработчик для кнопки публикации устройства
// Тексты сообщений для использования в JS
const messages = {
    busid_not_specified: "busid_not_specified",
    publishing: "Publishing...",
    publish_success: "publish_success",
    error: "Error",
    port_not_specified: "port_not_specified",
    refreshing: "refreshing",
    devices_refreshed: "devices_refreshed",
    busid: "Bus ID",
    device: "device",
    actions: "Actions",
    virtual: "virtual",
    publish: "Publish",
    set_name: "set_name",
    no_local_devices: "No local USB devices found",
    running_doctor: "Running diagnostics...",
    doctor_results: "Diagnostic Results"
};

document.querySelectorAll('.bind-device-btn').forEach(button => {
    button.addEventListener('click', function() {
        const busid = this.getAttribute('data-busid');
        if (!busid) {
            showNotification(messages.busid_not_specified, 'danger');
            return;
        }
        
        // Проверяем, является ли устройство виртуальным
        const isVirtual = this.hasAttribute('data-is-virtual');
        const virtualId = this.getAttribute('data-virtual-id');
        
        if (isVirtual && virtualId) {
            // Для виртуальных устройств открываем модальное окно подключения
            const modal = new bootstrap.Modal(document.getElementById('connectVirtualDeviceModal'));
            document.getElementById('virtual_device_id').value = virtualId;
            modal.show();
            return;
        }
        
        // Для реальных устройств отправляем запрос через USBIP
        
        // Меняем состояние кнопки
        const originalText = this.innerHTML;
        this.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>${messages.publishing}`;
        this.disabled = true;
        
        // Отправляем запрос на публикацию
        fetch('/bind_device', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
            },
            body: new URLSearchParams({
                'busid': busid
            })
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                showNotification(`${messages.publish_success} ${busid}`, 'success');
            } else {
                showNotification(`${messages.error}: ${data.message}`, 'danger');
            }
        })
        .catch(error => {
            showNotification(`${messages.error}: ${error}`, 'danger');
        })
        .finally(() => {
            // Восстанавливаем состояние кнопки
            this.innerHTML = originalText;
            this.disabled = false;
        });
    });
});

// Обработчик для кнопки отключения устройства
document.querySelectorAll('.detach-device-btn').forEach(button => {
    button.addEventListener('click', function() {
        const port = this.getAttribute('data-port');
        if (!port) {
            showNotification(messages.port_not_specified, 'danger');
            return;
        }
        
        // Проверяем, является ли порт виртуальным
        const isVirtual = this.hasAttribute('data-is-virtual');
        const virtualPortId = this.getAttribute('data-virtual-port-id');
        
        if (isVirtual && virtualPortId) {
            // Для виртуальных портов отправляем запрос на отключение виртуального устройства
            if (!confirm(`confirm_disconnect_virtual`)) {
                return;
            }
            
            // Меняем состояние кнопки
            const originalText = this.innerHTML;
            this.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>disconnecting`;
            this.disabled = true;
            
            // Отправляем форму на отключение виртуального устройства
            fetch('/disconnect_virtual_device', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: new URLSearchParams({
                    'port_id': virtualPortId
                })
            })
            .then(response => {
                if (response.ok) {
                    showNotification('virtual_device_disconnected', 'success');
                    // Перезагружаем страницу для обновления списка
                    window.location.reload();
                } else {
                    showNotification('virtual_device_disconnect_error', 'danger');
                    // Восстанавливаем состояние кнопки
                    this.innerHTML = originalText;
                    this.disabled = false;
                }
            })
            .catch(error => {
                showNotification(`${messages.error}: ${error}`, 'danger');
                // Восстанавливаем состояние кнопки
                this.innerHTML = originalText;
                this.disabled = false;
            });
            
            return;
        }
        
        // Для реальных устройств выполняем обычное отключение через USBIP
        if (!confirm(`confirm_disconnect ${port}?`)) {
            return;
        }
        
        // Меняем состояние кнопки
        const originalText = this.innerHTML;
        this.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>disconnecting`;
        this.disabled = true;
        
        // Отправляем запрос на отключение
        fetch('/detach_device', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
            },
            body: new URLSearchParams({
                'port': port
            })
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                showNotification(`Device disconnected from port ${port}`, 'success');
                // Удаляем строку из таблицы
                this.closest('tr').remove();
                // Если таблица пуста, показываем сообщение
                if (document.querySelectorAll('.attached-devices-container tbody tr').length === 0) {
                    document.querySelector('.attached-devices-container').innerHTML = `
                        <div class="alert alert-info m-3">
                            <i class="fas fa-info-circle me-2"></i>No remote devices attached
                        </div>
                    `;
                }
            } else {
                showNotification(`${messages.error}: ${data.message}`, 'danger');
            }
        })
        .catch(error => {
            showNotification(`${messages.error}: ${error}`, 'danger');
        })
        .finally(() => {
            // Восстанавливаем состояние кнопки
            this.innerHTML = originalText;
            this.disabled = false;
        });
    });
});

// Функция для обновления списка устройств через API
function refreshDevicesList() {
    // Меняем состояние кнопки
    const refreshBtn = document.querySelector('.refresh-devices-btn');
    const originalBtnText = refreshBtn.innerHTML;
    refreshBtn.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>${messages.refreshing || 'Обновление...'}`;
    refreshBtn.disabled = true;
    
    // Запрашиваем список устройств через API
    fetch('/api/devices/local')
        .then(response => {
            // Проверяем, успешен ли ответ
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            return response.json();
        })
        .then(data => {
            if (data.success) {
                // Обновляем содержимое таблицы
                const container = document.querySelector('.local-devices-container');
                if (data.devices && data.devices.length > 0) {
                    // Создаем HTML-таблицу
                    let html = `
                    <div class="table-responsive">
                        <table class="table table-hover mb-0">
                            <thead>
                                <tr>
                                    <th>${messages.busid || 'Bus ID'}</th>
                                    <th>${messages.device || 'Устройство'}</th>
                                    <th>${messages.actions || 'Действия'}</th>
                                </tr>
                            </thead>
                            <tbody>`;
                    
                    data.devices.forEach(device => {
                        const isVirtual = device.is_virtual || false;
                        const deviceName = device.device_name || device.info?.split(':')[1]?.trim() || 'Неизвестное устройство';
                        const vendorId = device.idVendor || device.vendor_id || '';
                        const productId = device.idProduct || device.product_id || '';
                        const busid = device.busid;
                        const virtualId = device.virtual_id || '';
                        
                        html += `
                        <tr ${isVirtual ? 'class="table-primary"' : ''}>
                            <td>
                                ${busid}
                                ${isVirtual ? `<span class="badge bg-primary">${messages.virtual || 'Виртуальное'}</span>` : ''}
                            </td>
                            <td>
                                <div>${deviceName}</div>
                                <small class="text-muted">${vendorId}:${productId}</small>
                            </td>
                            <td>
                                <div class="btn-group btn-group-sm">
                                    <button class="${device.is_published ? 'btn btn-success' : 'btn btn-primary'} bind-device-btn" 
                                            data-busid="${busid}"
                                            ${isVirtual ? `data-is-virtual="true" data-virtual-id="${virtualId}"` : ''}
                                            ${device.is_published ? 'disabled' : ''}>
                                        <i class="${device.is_published ? 'fas fa-check me-1' : 'fas fa-share me-1'}"></i>
                                        ${device.is_published ? (messages.published || 'Опубликовано') : (messages.publish || 'Опубликовать')}
                                    </button>
                                    ${!isVirtual ? `
                                    <button class="btn btn-outline-info set-alias-btn" 
                                            data-busid="${busid}" 
                                            data-device="${deviceName}">
                                        <i class="fas fa-tag me-1"></i>${messages.set_name || 'Задать имя'}
                                    </button>
                                    ` : ''}
                                </div>
                            </td>
                        </tr>`;
                    });
                    
                    html += `
                            </tbody>
                        </table>
                    </div>`;
                    
                    container.innerHTML = html;
                    
                    // Переназначаем обработчики событий для новых кнопок
                    attachBindButtonHandlers();
                    attachAliasButtonHandlers();
                    
                } else {
                    // Если устройств нет
                    container.innerHTML = `
                    <div class="alert alert-warning m-3">
                        <i class="fas fa-exclamation-triangle me-2"></i>${messages.no_local_devices || 'Нет локальных USB устройств'}
                        <div class="mt-2 small">
                            Возможные причины:
                            <ul class="mb-0">
                                <li>К компьютеру не подключены USB устройства</li>
                                <li>Служба usbipd не запущена</li>
                                <li>У приложения недостаточно прав для доступа к USB</li>
                            </ul>
                            <div class="mt-2">
                                Рекомендации:
                                <ul class="mb-0">
                                    <li>Подключите USB устройства к компьютеру</li>
                                    <li>Перезапустите службу: <code>sudo systemctl restart usbipd</code></li>
                                    <li>Запустите диагностику с помощью кнопки ниже</li>
                                </ul>
                            </div>
                            <div class="mt-3">
                                <button class="btn btn-warning run-doctor-btn">
                                    <i class="fas fa-stethoscope me-1"></i>Run Diagnostics
                                </button>
                            </div>
                        </div>
                    </div>`;
                    
                    // Добавляем обработчик для кнопки диагностики, т.к. она была добавлена динамически
                    const doctorBtn = container.querySelector('.run-doctor-btn');
                    if (doctorBtn) {
                        doctorBtn.addEventListener('click', runDoctorDiagnostics);
                    }
                }
                
                // Показываем уведомление об успешном обновлении
                showNotification(messages.devices_refreshed || `Список устройств обновлен: найдено ${data.devices.length} устройств`, 'success');
            } else {
                // Если произошла ошибка
                showNotification(`${messages.error || 'Ошибка'}: ${data.message}`, 'danger');
            }
        })
        .catch(error => {
            showNotification(`${messages.error || 'Ошибка'}: ${error}`, 'danger');
        })
        .finally(() => {
            // Восстанавливаем состояние кнопки
            refreshBtn.innerHTML = originalBtnText;
            refreshBtn.disabled = false;
        });
}

// Функция для прикрепления обработчиков к кнопкам публикации
function attachBindButtonHandlers() {
    document.querySelectorAll('.bind-device-btn').forEach(button => {
        button.addEventListener('click', function() {
            const busid = this.getAttribute('data-busid');
            if (!busid) {
                showNotification(messages.busid_not_specified, 'danger');
                return;
            }
            
            // Проверяем, является ли устройство виртуальным
            const isVirtual = this.hasAttribute('data-is-virtual');
            const virtualId = this.getAttribute('data-virtual-id');
            
            if (isVirtual && virtualId) {
                // Для виртуальных устройств открываем модальное окно подключения
                const modal = new bootstrap.Modal(document.getElementById('connectVirtualDeviceModal'));
                document.getElementById('virtual_device_id').value = virtualId;
                modal.show();
                return;
            }
            
            // Для реальных устройств отправляем запрос через USBIP
            
            // Меняем состояние кнопки
            const originalText = this.innerHTML;
            this.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>${messages.publishing}`;
            this.disabled = true;
            
            // Отправляем запрос на публикацию
            fetch('/bind_device', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: new URLSearchParams({
                    'busid': busid
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    showNotification(`${messages.publish_success} ${busid}`, 'success');
                    // Обновляем список устройств
                    refreshDevicesList();
                } else {
                    showNotification(`${messages.error}: ${data.message}`, 'danger');
                }
            })
            .catch(error => {
                showNotification(`${messages.error}: ${error}`, 'danger');
            })
            .finally(() => {
                // Восстанавливаем состояние кнопки
                this.innerHTML = originalText;
                this.disabled = false;
            });
        });
    });
}

// Функция для запуска диагностики
function runDoctorDiagnostics() {
    // Получаем кнопку, которая была нажата
    const button = this;
    
    // Показываем уведомление о запуске
    showNotification(messages.running_doctor, 'info');
    
    // Меняем состояние кнопки
    const originalText = button.innerHTML;
    button.innerHTML = `<i class="fas fa-spinner fa-spin me-1"></i>${messages.running_doctor}`;
    button.disabled = true;
    
    // Вызываем API для запуска doctor.sh
    fetch('/api/run_doctor', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        }
    })
    .then(response => response.json())
    .then(data => {
        // Восстанавливаем состояние кнопки
        button.innerHTML = originalText;
        button.disabled = false;
        
        if (data.success) {
            // Создаем модальное окно с результатами
            const modalId = 'doctorResultsModal';
            
            // Удаляем существующее модальное окно, если оно есть
            const existingModal = document.getElementById(modalId);
            if (existingModal) {
                existingModal.remove();
            }
            
            // Создаем новое модальное окно
            const modalHtml = `
            <div class="modal fade" id="${modalId}" tabindex="-1" aria-labelledby="${modalId}Label" aria-hidden="true">
                <div class="modal-dialog modal-lg modal-dialog-scrollable">
                    <div class="modal-content">
                        <div class="modal-header">
                            <h5 class="modal-title" id="${modalId}Label">${messages.doctor_results}</h5>
                            <button type="button" class="btn-close" data-bs-dismiss="modal" aria-label="Close"></button>
                        </div>
                        <div class="modal-body">
                            <pre class="bg-dark text-light p-3 rounded" style="max-height: 400px; overflow-y: auto;">${data.output}</pre>
                        </div>
                        <div class="modal-footer">
                            <button type="button" class="btn btn-secondary" data-bs-dismiss="modal">Закрыть</button>
                            <button type="button" class="btn btn-primary refresh-after-doctor">Обновить список устройств</button>
                        </div>
                    </div>
                </div>
            </div>`;
            
            // Добавляем модальное окно в DOM
            document.body.insertAdjacentHTML('beforeend', modalHtml);
            
            // Показываем модальное окно
            const modal = new bootstrap.Modal(document.getElementById(modalId));
            modal.show();
            
            // Добавляем обработчик для кнопки обновления списка устройств
            document.querySelector('.refresh-after-doctor').addEventListener('click', function() {
                modal.hide();
                refreshDevicesList();
            });
            
            // Обновляем список устройств
            refreshDevicesList();
        } else {
            showNotification(`${messages.error}: ${data.message}`, 'danger');
        }
    })
    .catch(error => {
        // Восстанавливаем состояние кнопки
        button.innerHTML = originalText;
        button.disabled = false;
        
        showNotification(`${messages.error}: ${error}`, 'danger');
    });
}
